[docs]    def extract_alert_and_hits_string_from_xml(
        self, xml_string: str
    ) -> Tuple[lxml.etree._Element, List[lxml.etree._Element]]:
        """Extracts alert and hits from xml.

        Args:
            xml_string (str): XML string that will be parsed.

        Returns:
            Tuple[lxml.etree._Element, List[lxml.etree._Element]]: Alert and hits.
        """
        xml_cleaned = self.clean_xml_declaration_regex.sub("", xml_string)
        alert = lxml.etree.fromstring(xml_cleaned)
        hits = alert.xpath('hits/elem[@name="hit"]')
        return alert, hits

[docs]    def remove_unnecessary_columns(self, df: pyspark.sql.DataFrame) -> pyspark.sql.DataFrame:
        df = (
            df.selectExpr("*", "alert_hits.*")
            .selectExpr("*", "explode(hits) as hit")
            .selectExpr("*", "alert_header.*")
            .selectExpr("*", "hit.*")
            .drop("alert_hits", "alert_header", "hits", "hit")
        )
        return df
